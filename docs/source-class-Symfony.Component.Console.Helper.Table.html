390:     private function buildTableRows($rows)
391:     {
392:         $unmergedRows = array();
393:         for ($rowKey = 0; $rowKey < count($rows); ++$rowKey) {
394:             $rows = $this->fillNextRows($rows, $rowKey);
395: 
396:             // Remove any new line breaks and replace it with a new line
397:             foreach ($rows[$rowKey] as $column => $cell) {
398:                 if (!strstr($cell, "\n")) {
399:                     continue;
400:                 }
401:                 $lines = explode("\n", $cell);
402:                 foreach ($lines as $lineKey => $line) {
403:                     if ($cell instanceof TableCell) {
404:                         $line = new TableCell($line, array('colspan' => $cell->getColspan()));
405:                     }
406:                     if (0 === $lineKey) {
407:                         $rows[$rowKey][$column] = $line;
408:                     } else {
409:                         $unmergedRows[$rowKey][$lineKey][$column] = $line;
410:                     }
411:                 }
412:             }
413:         }
414: 
415:         $tableRows = array();
416:         foreach ($rows as $rowKey => $row) {
417:             $tableRows[] = $this->fillCells($row);
418:             if (isset($unmergedRows[$rowKey])) {
419:                 $tableRows = array_merge($tableRows, $unmergedRows[$rowKey]);
420:             }
421:         }
422: 
423:         return $tableRows;
424:     }

426:     /**
427:      * fill rows that contains rowspan > 1.
428:      *
429:      * @param array $rows
430:      * @param int   $line
431:      *
432:      * @return array
433:      */
434:     private function fillNextRows($rows, $line)
435:     {
436:         $unmergedRows = array();
437:         foreach ($rows[$line] as $column => $cell) {
438:             if ($cell instanceof TableCell && $cell->getRowspan() > 1) {
439:                 $nbLines = $cell->getRowspan() - 1;
440:                 $lines = array($cell);
441:                 if (strstr($cell, "\n")) {
442:                     $lines = explode("\n", $cell);
443:                     $nbLines = count($lines) > $nbLines ? substr_count($cell, "\n") : $nbLines;
444: 
445:                     $rows[$line][$column] = new TableCell($lines[0], array('colspan' => $cell->getColspan()));
446:                     unset($lines[0]);
447:                 }
448: 
449:                 // create a two dimensional array (rowspan x colspan)
450:                 $unmergedRows = array_replace_recursive(array_fill($line + 1, $nbLines, ''), $unmergedRows);
451:                 foreach ($unmergedRows as $unmergedRowKey => $unmergedRow) {
452:                     $value = isset($lines[$unmergedRowKey - $line]) ? $lines[$unmergedRowKey - $line] : '';
453:                     $unmergedRows[$unmergedRowKey][$column] = new TableCell($value, array('colspan' => $cell->getColspan()));
454:                 }
455:             }
456:         }
457: 
458:         foreach ($unmergedRows as $unmergedRowKey => $unmergedRow) {
459:             // we need to know if $unmergedRow will be merged or inserted into $rows
460:             if (isset($rows[$unmergedRowKey]) && is_array($rows[$unmergedRowKey]) && ($this->getNumberOfColumns($rows[$unmergedRowKey]) + $this->getNumberOfColumns($unmergedRows[$unmergedRowKey]) <= $this->numberOfColumns)) {
461:                 foreach ($unmergedRow as $cellKey => $cell) {
462:                     // insert cell into row at cellKey position
463:                     array_splice($rows[$unmergedRowKey], $cellKey, 0, array($cell));
464:                 }
465:             } else {
466:                 $row = $this->copyRow($rows, $unmergedRowKey - 1);
467:                 foreach ($unmergedRow as $column => $cell) {
468:                     if (!empty($cell)) {
469:                         $row[$column] = $unmergedRow[$column];
470:                     }
471:                 }
472:                 array_splice($rows, $unmergedRowKey, 0, array($row));
473:             }
474:         }
475: 
476:         return $rows;
477:     }

488:         $newRow = array();
